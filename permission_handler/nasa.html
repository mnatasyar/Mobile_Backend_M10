void simulateMicrophoneFunctionality() async { var status = await Permission.microphone.status; if (status.isGranted) { setState(() { isMicrophoneAllowed = true; }); } else { status = await Permission.microphone.request(); if (status.isGranted) { setState(() { isMicrophoneAllowed = true; }); } else if (status.isPermanentlyDenied) { openAppSettings(); } } print("Simulating microphone functionality..."); }
